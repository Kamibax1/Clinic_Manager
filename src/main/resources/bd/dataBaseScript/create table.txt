-- create table

create table Role(
    id_Role serial primary key,
    name varchar(50) not null
);

create table Clinic_User(
    id_User serial primary key,
    email varchar(100) not null,
    password varchar(100) not null,
    created_at timestamp default current_timestamp,
    updated_at timestamp default current_timestamp,
    id_Role int not null references Role (id_Role)

);

create table Doctor(
    id_Doctor serial primary key,
    first_name varchar(50) not null,
    last_name varchar(50) not null,
    experience_years int not null,
    created_at timestamp default current_timestamp,
    updated_at timestamp default current_timestamp,
    id_User int not null references Clinic_User (id_User)
);

create table Patient(
    id_Patient serial primary key,
    full_name varchar(100) not null,
    date_birth varchar(50) not null,
    gender varchar(10) not null,
    phone_number varchar(50) not null,
    created_at timestamp default current_timestamp,
    updated_at timestamp default current_timestamp,
    id_User int not null references Clinic_User (id_User)
);

create table Specialization(
    id_Specialization serial primary key,
    name varchar(50) not null
);

create table Doctor_Specialization(
    id_Doctor_Specialization serial primary key,
    id_Doctor int not null references Doctor (id_Doctor),
    id_Specialization int not null references Specialization (id_Specialization)
);

create table Appointment(
    id_Appointment serial primary key,
    appointment_date date not null,
    status varchar(50) not null,
    symptoms text not null,
    created_at timestamp default current_timestamp,
    updated_at timestamp default current_timestamp,
    id_Doctor_Specialization int not null references Doctor_Specialization (id_Doctor_Specialization),
    id_Patient int not null references Patient (id_Patient)
);
